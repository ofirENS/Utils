1function vals = Bcoeff(ksi,j,x,t,gridT)
 2% evaluate the coefficient B at the  boundary ksi=0 or ksi=1;
 3% for the index j which describes the time steps timePoints_j, at time t and space
 4% point x
 5% timePoints is a vector describing the time descritized domain
 6
 7if t<=gridT(j)
 8    vals = 0;
 9elseif all([t>gridT(j), t<= gridT(j+1),x==ksi])
10    vals = 0;
11elseif all([t>gridT(j), t<=gridT(j+1),x~=ksi])
12    z0   = abs(x-ksi)/(2*sqrt(t-gridT(j)));
13    vals = -(erfc(z0))/2;
14elseif t>gridT(j+1)
15    z0   = abs(x-ksi)/(2*sqrt(t-gridT(j)));
16    z1   = abs(x-ksi)/(2*sqrt(t-gridT(j+1)));
17    vals = (erf(z0)-erf(z1))/2;
18else 
19   vals = 0; 
20end
21end
