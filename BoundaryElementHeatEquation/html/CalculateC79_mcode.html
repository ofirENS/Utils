1function [C] = CalculateC(N,N0,t,gridX)
 2% calculate the C matrix
 3% N- number of time points
 4% N0- number of space points
 5% t- time points
 6% gridX - grid for the space points
 7C = zeros(2*N,N0);
 8for i = 1:N% space
 9    for k=1:N0% time
10        C((2*i-1):2*i,k) = [Ccoeff(k,0,t(i),gridX);
11                            Ccoeff(k,1,t(i),gridX)];
12    end
13end
