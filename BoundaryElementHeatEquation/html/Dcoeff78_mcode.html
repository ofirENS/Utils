1function vals = Dcoeff(y,j,x,t,gridT,f)
 2% Calculate the integral int_0^1(f(y,t_j)*A_{yj}(x,t)dy) by midpoint
 3% integration rule
 4% the spatial integration of all points y at time t_j
 5a    = zeros(1,numel(y));
 6vals = 0;
 7a(1) = Acoeff(y(1),j,x,t,gridT);
 8% y = 0:1/(numel(y)-2):1;
 9for aIdx = 2:numel(f);
10    a(aIdx) = Acoeff(y(aIdx),j,x,t,gridT);
11    vals    = vals+ 0.5*(a(aIdx)*f(aIdx)+a(aIdx-1)*f(aIdx-1))*(y(aIdx)-y(aIdx-1));
12end
13end
