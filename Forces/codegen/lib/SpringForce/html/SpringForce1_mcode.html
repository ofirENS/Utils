1function force = SpringForce(particleDist,springConst,connectivityMap,minParticleDist,fixedParticleNum)
 2% Calculate the spring force between N particles in any dimension M.
 3% particleDist    - NxN matrix of pairwise particle distances
 4% springConst     - NxN double matrix of spring constants
 5% connectivityMap - NxN binary matrix which defines the connectivity between particles
 6% minParticleDist - minimal distance between particles
 7% fixedParticleNum - particles in the system which do not move
 8
 9connectivityMap          = double(connectivityMap);
10% force              = springConst.*particleDist.*connectivityMap;
11particleDist = particleDist+diag(Inf*ones(1,size(particleDist,1)));
12L                         = (1-minParticleDist./particleDist).*connectivityMap;
13sumForces                 = sum(L,2);
14force                     = springConst.*(diag(sumForces)-L); % set the maindiagonal
15force(fixedParticleNum,:) = 0;% zero out forces for fixed particles
16force(:,fixedParticleNum) = 0;% zero out forces for fixed particles
17
