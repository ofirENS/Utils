1function [force, forceDirection] = LennardJones(particlePosition,particleDist,LJPotentialWidth,LJPotentialDepth,potentialType)
 2   % Calculate the lennard jones potential between beads and its derivative
 3   % particlePosition is a 3 by n matrix of particle position in space 
 4   % particle dist is an n by n pairwise distance matrix 
 5   % LJpotentialwidth is a scalar width of the LJ
 6   % LJPotentialepth is a scalar depth of the LJ
 7   % potentialType is a string stating the type of potential 'attractive' |
 8   % 'repulsive', or 'full'
 9   
10%    if ~exist('PotentialType','var')
11%        potentialType = 'full';
12%    end
13    numParticles = size(particlePosition,1);
14    dimension    = size(particlePosition,2);
15    force        = zeros(numParticles,dimension);
16    beadsDistMat = particleDist;
17    rm           = LJPotentialWidth; % the distance in which the potential is minimal
18    epsilon      = LJPotentialDepth;
19    bdmInv       = (beadsDistMat).^(-1);   % one over the bead distance matrix
20    % remove the diagonal 
21    bdmInv = bdmInv-diag(diag(bdmInv));
22    
23    % d            = MatIntPower(bdmInv, 6); % matrix integer power (mex form Utils)
24    % Raise bdInv to the sixt power
25    nb = dec2bin(6)-'0';
26    ak = bdmInv;
27    if nb(numel(nb))
28        t = bdmInv;
29    else
30        t = ones(size(ak));
31    end
32    
33    inds = nb(end-1:-1:1);
34    for nbk = 1:numel(inds)
35        ak = ak.*ak;
36        if inds(nbk)
37            t = t.*ak;
38        end
39    end
40     
41    A            = (rm^6).*t;
42%     forceValue   = 24*(epsilon*bdmInv).*(-2*t.*t +t); % derivative of LJ function
43    
44    forceValue   = (2*epsilon*rm^6).*(A-1);
45    
46    if strcmpi(potentialType,'repulsive')
47        forceValue(particleDist<2.5*rm)=0;% truncate
48    elseif strcmpi(potentialType','attractive')
49        forceValue(particleDist>2.5*rm)=0;% truncate
50    end
51    
52    forceValue(isnan(forceValue))= 0;
53    forceDirection = zeros(numParticles,numParticles,dimension);
54    % sum the contribution from all particles
55    for dIdx = 1:dimension
56        % replicate the position vector
57        A    = particlePosition(:,dIdx);
58        siz  = [1,numParticles];
59        B1   = A(:,ones(siz(2),1));
60        siz  = [numParticles,1];
61        A    = A';
62        B2   = A(ones(siz(1),1),:);
63        % Subtract positions to get the direction vectors
64        forceDirection(:,:,dIdx)  = B1-B2;
65        force(:,dIdx) = (sum(forceDirection(:,:,dIdx).*forceValue,1)');
66    end
67end
