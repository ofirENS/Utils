1function [force, forceDirection] = LennardJones(particlePosition,particleDist,LJPotentialWidth,LJPotentialDepth)
 2   % Calculate the lennard jones potential between beads and its derivative
 3   % 
 4    numParticles = size(particlePosition,1);
 5    dimension    = size(particlePosition,2);
 6    force        = zeros(numParticles,dimension);
 7    beadsDistMat = particleDist;
 8    rm           = LJPotentialWidth; % the distance in which the potential is minimal
 9    epsilon      = LJPotentialDepth;
10    bdmInv       = (beadsDistMat).^(-1);   % one over the bead distance matrix
11    % remove the diagonal 
12    bdmInv = bdmInv-diag(diag(bdmInv));
13    
14    % d            = MatIntPower(bdmInv, 6); % matrix integer power (mex form Utils)
15    % Raise bdInv to the sixt power
16    nb = dec2bin(6)-'0';
17    ak = bdmInv;
18    if nb(numel(nb))
19        t = bdmInv;
20    else
21        t = ones(size(ak));
22    end
23    
24    inds = nb(end-1:-1:1);
25    for nbk = 1:numel(inds)
26        ak = ak.*ak;
27        if inds(nbk)
28            t = t.*ak;
29        end
30    end
31     
32    A            = (rm^6).*t;
33%     forceValue   = 24*(epsilon*bdmInv).*(-2*t.*t +t); % derivative of LJ function
34    
35    forceValue   = (2*epsilon*rm^6).*(A-1);
36    forceValue(isnan(forceValue))= 0;
37    forceDirection = zeros(numParticles,numParticles,dimension);
38    % sum the contribution from all particles
39    for dIdx = 1:dimension
40        % replicate the position vector
41        A    = particlePosition(:,dIdx);
42        siz  = [1,numParticles];
43        B1   = A(:,ones(siz(2),1));
44        siz  = [numParticles,1];
45        A    = A';
46        B2   = A(ones(siz(1),1),:);
47        % Subtract positions to get the direction vectors
48        forceDirection(:,:,dIdx)  = B2-B1;
49        force(:,dIdx) = (sum(forceDirection(:,:,dIdx).*forceValue,1)');
50    end
51end
