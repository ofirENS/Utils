1function force = MechanicalSpringPointForce(particlePosition,pointSourcePosition, forceDirection, forceMagnitude,cutoff)
 2
 3% Apply an harmonic pushing force for particles around force sources. 
 4
 5% Calculate the distance betwen each particle and the point source position
 6force = zeros(size(particlePosition,1),3);
 7for sIdx = 1:size(pointSourcePosition,1)    
 8      forceDir     = 2*bsxfun(@minus,particlePosition,pointSourcePosition(sIdx,:));
 9      distToSource = cutoff^2- sum(forceDir.^2,2);
10      distToSource(distToSource<0) = 0;
11      force  = force + bsxfun(@times,distToSource,forceDir);
12end
13if forceDirection==-1
14    force = -forceMagnitude*force;
15elseif forceDirection==1
16    force = forceMagnitude*force;
17else 
18    error('unsupported direction option')
19end
