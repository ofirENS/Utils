1function [force] = MorseForce(De,alpha, re, edgesX,edgesY,edgesZ, particleDistance,forceType)
 2% calculate the Morse force, derived from the Morse potential
 3% u = De*(exp(-2*a(r-re)) -2*exp(-a*(r-re)))
 4% The force is given by 
 5% F = dv/dr = De*(-2*a*exp(-2*a*(r-re)) +2*a*exp(-a*(r-re)))
 6% Input parameters:
 7% re - the equilibrium bond distance 
 8% De - the well depth
 9% alpha - the width of the potential
10% edgesX, edgesY, edgesZ -  are numPartXnumPart matrices representing the vectors
11% between particles in the X YT and Z respectively.
12% particleDistance -  a numPartXnumPart matrix of pairwise distances 
13% forceType is a string [attractive |repuslsive |full] indicating whether
14% to use the attractive repulsive part of the Morse force or both
15% For forces in 1 or 2D set the position of the particle in 2-3 dimension
16% to be constant (preferably 0) prior to using this function 
17
18% edgesVector(:,:,1)  = edgesX;
19% edgesVector(:,:,2)  = edgesY;
20% edgesVector(:,:,3)  = edgesZ;
21
22dim = 3; 
23% dim                 = size(edgesVector,3);
24numParticles        = size(particleDistance,1);
25
26% Calculate the force magnitude
27repulse      = exp(-alpha*(particleDistance-re)); % repulsive term
28attract      = 1-repulse; % attractive term 
29if strcmpi(forceType,'attractive')
30    forceMag = 2*alpha*De*attract;
31elseif strcmpi(forceType,'repulsive')
32    forceMag = 2*alpha*De*repulse;
33elseif strcmpi(forceType,'full');
34    forceMag = 2*De*alpha*attract*repulse;
35else
36    error('forceType not supported')
37end
38
39
40% remove the diagonal 
41% Normalize the mean direction vectors to unit vectors
42dirNorm = sqrt(edgesX.^2+edgesY.^2+edgesZ.^2);
43dirNorm = dirNorm+eye(numParticles);% put ones in the diagonal to prevent dividing by zeros
44dirVecX = edgesX./dirNorm;
45dirVecY = edgesY./dirNorm;
46dirVecZ = edgesZ./dirNorm;
47
48% dirVec  = bsxfun(@rdivide, edgesVector,dirNorm);
49
50% Multiply by the force magnitude
51forceVecX  = forceMag.*dirVecX;
52forceVecY  = forceMag.*dirVecY;
53forceVecZ  = forceMag.*dirVecZ;
54
55% bsxfun(@times,forceMag,dirVec);
56
57% Calculate the mean force direction and assign it to each particle
58force      = zeros(numParticles,dim);
59force(:,1) = -sum(forceVecX,2);
60force(:,2) = -sum(forceVecY,2);
61force(:,3) = -sum(forceVecZ,2);
62
63
64% for dIdx = 1:dim
65%  force(:,dIdx) = -sum(forceVec(:,:,dIdx),2);
66% end
