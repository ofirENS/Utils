1function edgesVec   = GetEdgesVectors(particlePosition, connectivityMap)
 2    % Calculate the edges vectors between connected particles
 3    % particlePosition is an NxDim vector of position
 4    % connectivityMat is an NxN logical matrix of particle connectivity
 5    edgesVec = zeros(size(particlePosition,1),size(particlePosition,1), size(particlePosition,2));
 6    for dIdx=1:size(particlePosition,2)
 7        edgesVec(:,:,dIdx) = bsxfun(@times,particlePosition(:,dIdx),connectivityMap)-...
 8            bsxfun(@times,particlePosition(:,dIdx)',connectivityMap);
 9    end
10end
