1function force = LennardJones(particlePosition,particleDist,LJPotentialWidth,LJPotentialDepth)
 2    numParticles = size(particlePosition,1);
 3    dimension    = size(particlePosition,2);
 4    force        = zeros(numParticles,dimension);
 5    beadsDistMat = particleDist;
 6    sig          = LJPotentialWidth;
 7    epsilon      = LJPotentialDepth;
 8    bdmInv       = (beadsDistMat).^(-1);   % one over the bead distance matrix
 9
10    %                 d            = MatIntPower(bdmInv, 6); % matrix integer power (mex form Utils)
11
12    nb = dec2bin(6)-'0';
13    ak = bdmInv;
14    if nb(numel(nb))
15        d = bdmInv;
16    else
17        d = ones(size(ak));
18    end
19    
20    inds = nb(end-1:-1:1);
21    for nbk = 1:numel(inds)
22        ak = ak.*ak;
23        if inds(nbk)
24            d = d.*ak;
25        end
26    end
27
28    t            = (sig^6).*d;
29    forceValue   = 24*(epsilon*bdmInv).*(-2*t.*t +t); % derivative of LJ function
30
31    forceValue(isnan(forceValue))= 0;
32    for dIdx = 1:dimension
33        % replicate the position vector
34        A    = particlePosition(:,dIdx);
35        siz  = [1,numParticles];
36        B1   = A(:,ones(siz(2),1));
37        siz  = [numParticles,1];
38        A    = A';
39        B2   = A(ones(siz(1),1),:);
40        % Subtract positions to get the direction vectors
41        fd  = B1-B2;
42        force(:,dIdx) = (sum(fd.*forceValue,1)');
43    end
44end
